3.0动态sql

动态sql就是根据需要动态生成SQL
！任何没有使用动态SQL标签的语句都需要当心它潜在的问题

#if
如果满足if中的条件，则添加标签中的SQL
//if中的条件，默认获取对象中的属性，比如在入参对象user中的name属性
//如果入参是一个基本类型，比如说String，那么必须指定别名
<select id="queryStuByConditionIfWhere" resultType="student">
    select <include refid="prStu"/>
    from students
        where grade_id =1
        <if test="stuNo != null">
            and stu_no stu_no = #{stuNo}
        </if>
        <if test="stuName != null">
            and stu_name =#{stuName}
        </if>
</select>
if的缺点：如果前面没有where条件，那么当前if中的and就是多余的

#where
通常配合if使用，可以屏蔽第一个多出来的and或者or
//语句后面多出来的无法屏蔽
<select id="queryStuByConditionIfWhere" resultType="student">
    select <include refid="prStu"/>
    from students
    <where>
        //where中也可以添加自定义sql，一般为固定条件
        grade_id =1
        <if test="stuNo != null">
            and stu_no stu_no = #{stuNo}
        </if>
        <if test="stuName != null">
            and stu_name =#{stuName}
        </if>
    </where>
</select>


#trim
可以在条件判断之后的SQL语句前后，添加或者去掉指定的字符
prefix：添加前缀（不一定是where，写什么就是什么）
suffix：添加后缀
prefixOverrides：去除指定多余的前缀
suffixOverrides：去除指定多余的后缀
  • <trim prefix="where" prefixOverrides="and|or">相当于where标签
<select id="queryStuByConditionTrim" resultType="student">
    select <include refid="prStu"/>
    from students
    <trim prefix="where" prefixOverrides="and |or " suffixOverrides="and |or ">
        <if test="stuNo != null">
            and stu_no  = #{stuNo} and
        </if>
        <if test="stuName != null">
            and stu_name =#{stuName} and
        </if>
    </trim>
</select>

  • <trim prefix="set" suffixOverrides=",">相当于set标签
<update id="updateStuByConditionTrim">
    update students
    <trim prefix="set" suffixOverrides=",">
        <if test="stuName != null">
            stu_name =#{stuName},
        </if>
        <if test="email != null">
            email = #{email},
        </if>
        <if test="phone != null">
            phone = #{phone},
        </if>
    </trim>
    where stu_no = #{stuNo}
</update>

#set
用于更新语句中去除后面的多逗号
<update id="updateStuByConditionSet">
    update students
    <set>
        <if test="stuName != null">
            stu_name =#{stuName},
        </if>
        <if test="email != null">
            email = #{email},
        </if>
        <if test="phone != null">
            phone = #{phone},
        </if>
    </set>
        where stu_no = #{stuNo}
</update>


#forEach
用于遍历list类型的条件
  • collection：要迭代的集合对象
  • item：迭代的元素（别名）
  • open：开始时添加的内容
  • close：结束时添加的呢绒
  • separator：每个元素间添加的内容
  • index：起始位置
查询演示：
<select id="queryStudentByConditionForeach" resultType="com.jac.beans.Student">
    select <include refid="prStu"/>
    from students
    where stu_no in
    //这一句话完全可以写在open属性中，或者使用trim标签替代
    <foreach collection="condition" item="stuNo" open="(" close=")" separator=",">
        #{stuNo}
    </foreach>
</select>
插入演示：
<insert id="addStuConditionBatch">
    insert into students
    (stu_name,grade_id,sex,stu_pwd)
    VALUES
    <foreach collection="students" item="stu" separator=",">
        (#{stu.stuName},#{stu.gradeId},#{stu.sex},#{stu.stuPwd})
    </foreach>
</insert>


#bind
bind用于存储一些结算后的值
//mybatis的表达式如果要进行计算操作，只能用${}，所以需要bind替代它避免非预编译语句的产生
例：分页查询
<select id="queryStuByConditionPage" resultType="com.jac.beans.Student">
    <bind name="fromIndex" value="(pageNo-1)*pageSize"/>
    select <include refid="prStu"/>
    from students
    <where>
        <if test="student.stuName != null">
            and stu_name like concat('%',#{student.stuName},'%')
        </if>
        <if test="student.gradeId != null">
            and grade_id = #{student.gradeId}
        </if>
    </where>
    order by stu_no
    limit #{fromIndex}, #{pageSize}
</select>
