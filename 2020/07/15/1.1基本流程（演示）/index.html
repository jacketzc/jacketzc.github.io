1.1基本流程（演示）

Elasticsearch和非关系型数据库类似，存数据的时候不需要表，也不需要管理这些表
#索引（名词）
相当于一个数据库
#索引（动词）
索引一个文档，就是存储一个文档到一个索引（名词）
//相当于insert

以下操作全部使用省略格式（省略主机名等，只保留请求路径以及参数）

#插入（PUT）
//更新也可以使用它
PUT /megacorp/employee/1
{
    "first_name" : "John",
    "last_name" :  "Smith",
    "age" :        25,
    "about" :      "I love to go rock climbing",
    "interests": [ "sports", "music" ]
}
返回：
  • _index：索引名称
  • _type：类型名称
  • _id：该条插入的数据的id（）
{
    "_index": "megacory",
    "_type": "employee",
    "_id": "1",
    "_version": 1,
    "result": "created",
    "_shards": {
        "total": 2,
        "successful": 1,
        "failed": 0
    },
    "_seq_no": 0,
    "_primary_term": 1
}


#获取、搜索（根据id）（GET）
//和PUT一样，省去了<BODY>部分
GET /megacorp/employee/1

#验证存在与否（HEAD）
HEAD /megacorp/employee/1

#搜索
  • 简单搜索
    • 搜索所有
GET /megacorp/employee/_search
    • 根据其中的某一条属性
GET /megacorp/employee/_search?q=last_name:Smith
  • 使用查询表达式（JSON）
GET /megacorp/employee/_search
{
    "query" : {
        "match" : {
            "last_name" : "Smith"
        }
    }
}
    • 更复杂的搜索（使用过滤器）
//匹配年龄大于30的文档
GET /megacorp/employee/_search
{
    "query" : {
        "bool": {
            "must": {
                "match" : {
                    "last_name" : "smith"
                }
            },
            "filter": {
                "range" : {
                    "age" : { "gt" : 30 }
                }
            }
        }
    }
}
  • 全文搜索
//Elasticsearch会将提供的条件拆分进行搜索
//Elasticsearch 默认按照相关性得分排序
GET /megacorp/employee/_search
{
    "query" : {
        "match" : {
            "about" : "rock climbing"
        }
    }
}
结果：
{
    "took": 7,
    "timed_out": false,
    "_shards": {
        "total": 1,
        "successful": 1,
        "skipped": 0,
        "failed": 0
    },
    "hits": {
        "total": {
            "value": 2,
            "relation": "eq"
        },
        "max_score": 1.4167401,
        "hits": [
            {
                "_index": "megacory",
                "_type": "employee",
                "_id": "1",
                "_score": 1.4167401,
                "_source": {
                    "first_name": "John",
                    "last_name": "Smith",
                    "age": 25,
                    "about": "I love to go rock climbing",
                    "interests": [
                        "sports",
                        "music"
                    ]
                }
            },
            {
                "_index": "megacory",
                "_type": "employee",
                "_id": "2",
                "_score": 0.4589591,
                "_source": {
                    "first_name": "Jane",
                    "last_name": "Smith",
                    "age": 32,
                    "about": "I like to collect rock albums",
                    "interests": [
                        "music"
                    ]
                }
            }
        ]
    }
}
  • 精确短语搜索
//相对于全文搜索，Elasticsearch将不再拆分短语
GET /megacorp/employee/_search
{
    "query" : {
        "match_phrase" : {
            "about" : "rock climbing"
        }
    }
}
  • 高亮搜索
Elasticsearch可以指定搜素结果中加以强调的部分
//为强调的部分添加<em></em>进行封装
GET /megacorp/employee/_search
{
    "query" : {
        "match_phrase" : {
            "about" : "rock climbing"
        }
    },
    "highlight": {
        "fields" : {
            "about" : {}
        }
    }
}
结果：
{
   ...
   "hits": {
      "total":      1,
      "max_score":  0.23013961,
      "hits": [
         {
            ...
            "_score":         0.23013961,
            "_source": {
               "first_name":  "John",
               "last_name":   "Smith",
               "age":         25,
               "about":       "I love to go rock climbing",
               "interests": [ "sports", "music" ]
            },
            "highlight": {
               "about": [
                  "I love to go <em>rock</em> <em>climbing</em>"
               ]
            }
         }
      ]
   }
}
  • 聚合搜索
Elasticsearch可以对搜索到的数据进行分析
